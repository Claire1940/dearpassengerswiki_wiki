  <Step num="1" title="Аналіз польотного маніфесту">

  </Step>
  <Step num="3" title="Налагодження зв'язку з пілотом">
    Пілот повинен попереджати екіпаж салону перед входом у зону динамічної погоди або проходженням повітряних ям. Це попередження дозволяє відповідальному за вантаж притримати нестабільні ящики.
  </Step>
  <Step num="4" title="Управління зміщеннями під час польоту">

  </Step>
  <Step num="5" title="Негайне усунення проривів">

  </Step>
</Steps>
